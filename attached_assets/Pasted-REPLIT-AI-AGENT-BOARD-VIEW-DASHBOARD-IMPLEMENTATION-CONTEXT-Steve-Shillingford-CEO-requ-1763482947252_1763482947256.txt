## PART 7: DATA CATEGORIZATION LOGIC

**How to Categorize Features into 3 Pillars:**

```typescript
interface FeatureCategorization {
  pillar: 'revenue' | 'operations' | 'innovation';
  isPrimary: boolean; // Some features can be in multiple pillars
}

function categorizeFeature(feature: Feature): FeatureCategorization[] {
  const categories: FeatureCategorization[] = [];

  // PILLAR 1: REVENUE GROWTH
  // Features that directly generate new ARR or expand existing contracts
  if (
    feature.arr_impact > 0 && 
    !feature.is_artemis_work &&
    (feature.priority_tier === 0 || feature.priority_tier === 1)
  ) {
    categories.push({ pillar: 'revenue', isPrimary: true });
  }

  // PILLAR 2: OPERATIONAL EXCELLENCE
  // Features that protect ARR, fix bugs, improve performance, or ensure compliance
  if (
    feature.priority_tier === 0 || // Emergency fixes
    feature.feature_name.toLowerCase().includes('bug') ||
    feature.feature_name.toLowerCase().includes('fix') ||
    feature.feature_name.toLowerCase().includes('performance') ||
    feature.feature_name.toLowerCase().includes('security') ||
    feature.feature_name.toLowerCase().includes('monitoring') ||
    feature.feature_name.toLowerCase().includes('compliance') ||
    feature.feature_name.toLowerCase().includes('stability')
  ) {
    categories.push({ pillar: 'operations', isPrimary: true });
  }

  // PILLAR 3: INNOVATION/COMPETITIVE ADVANTAGE
  // Features marked as Artemis work or strategic platform development
  if (
    feature.is_artemis_work === true ||
    feature.artemis_category !== null ||
    feature.feature_name.toLowerCase().includes('artemis') ||
    feature.feature_name.toLowerCase().includes('platform') ||
    feature.feature_name.toLowerCase().includes('information graph') ||
    feature.feature_name.toLowerCase().includes('service fabric')
  ) {
    categories.push({ pillar: 'innovation', isPrimary: true });
  }

  // If no categorization, default to operations (maintenance work)
  if (categories.length === 0) {
    categories.push({ pillar: 'operations', isPrimary: false });
  }

  return categories;
}

// Usage
const allFeatures = features.map(f => ({
  ...f,
  pillars: categorizeFeature(f)
}));

// Get features for specific pillar
const revenueFeatures = allFeatures.filter(f => 
  f.pillars.some(p => p.pillar === 'revenue' && p.isPrimary)
);
```

```typescript
// Revenue Metrics
function calculateRevenueMetrics(features: Feature[], pipeline: SalesOpportunity[]) {
  const revenueFeatures = features.filter(f => f.arr_impact > 0 && !f.is_artemis_work);
  
  return {
    totalARR: pipeline.reduce((sum, p) => sum + (p.arr || 0), 0),
    newClients: [...new Set(pipeline.filter(p => p.type === 'New Business').map(p => p.client_name))].length,
    avgDealSize: pipeline.reduce((sum, p) => sum + (p.arr || 0), 0) / pipeline.length,
    conversionRate: (pipeline.filter(p => p.pipeline_stage === 'Closed Won').length / pipeline.length * 100).toFixed(0),
    salesVelocity: calculateAvgSalesCycle(pipeline),
    topFeatures: revenueFeatures.sort((a, b) => b.arr_impact - a.arr_impact),
    completedFeatures: revenueFeatures.filter(f => f.status === 'Done').length,
    totalFeatures: revenueFeatures.length,
    completionRate: (revenueFeatures.filter(f => f.status === 'Done').length / revenueFeatures.length * 100).toFixed(0),
    growthRate: 15 // Calculate QoQ growth from historical data
  };
}

// Operations Metrics
function calculateOperationsMetrics(features: Feature[], tickets: JiraTicket[]) {
  const operationalFeatures = features.filter(f => 
    f.priority_tier === 0 || 
    f.feature_name.includes('Bug') || 
    f.feature_name.includes('Performance')
  );
  
  return {
    arrAtRisk: operationalFeatures
      .filter(f => f.priority_tier === 0)
      .reduce((sum, f) => sum + (f.arr_impact || 0), 0),
    criticalIssues: operationalFeatures.filter(f => f.priority_tier === 0).length,
    uptime: 99.7, // From monitoring system
    mttr: 4.2, // Mean time to resolution in hours
    costAvoidance: 150000, // Estimated from tech debt reduction
    criticalFeatures: operationalFeatures.sort((a, b) => a.priority_tier - b.priority_tier),
    resolvedIssues: operationalFeatures.filter(f => f.status === 'Done').length,
    totalIssues: operationalFeatures.length,
    completionRate: (operationalFeatures.filter(f => f.status === 'Done').length / operationalFeatures.length * 100).toFixed(0)
  };
}

// Innovation Metrics
function calculateInnovationMetrics(features: Feature[]) {
  const artemisFeatures = features.filter(f => f.is_artemis_work === true);
  const totalEffort = features.reduce((sum, f) => sum + (f.effort_estimate_weeks || 0), 0);
  const artemisEffort = artemisFeatures.reduce((sum, f) => sum + (f.effort_estimate_weeks || 0), 0);
  
  return {
    artemisPercentage: (artemisEffort / totalEffort * 100).toFixed(1),
    engineeringWeeks: artemisEffort,
    strategicFeatures: artemisFeatures.length,
    platformSplit: (artemisEffort / totalEffort * 100).toFixed(0),
    artemisFeatures: artemisFeatures.sort((a, b) => (a.target_quarter || '').localeCompare(b.target_quarter || '')),
    completedFeatures: artemisFeatures.filter(f => f.status === 'Done').length,
    totalFeatures: artemisFeatures.length,
    completionRate: (artemisFeatures.filter(f => f.status === 'Done').length / artemisFeatures.length * 100).toFixed(0)
  };
}

// Risk Level Classification
function getRiskLevel(arrAtRisk: number): 'low' | 'medium' | 'high' | 'critical' {
  if (arrAtRisk === 0) return 'low';
  if (arrAtRisk < 100000) return 'medium';
  if (arrAtRisk < 500000) return 'high';
  return 'critical';
}

// Replicability Class
function getReplicabilityClass(replicability: number): string {
  if (replicability >= 4) return 'high';
  if (replicability >= 3) return 'medium';
  return 'low';
}

// Revenue by Client
function getRevenueByClient(pipeline: SalesOpportunity[]) {
  const clientMap = new Map<string, number>();
  
  pipeline.forEach(p => {
    const current = clientMap.get(p.client_name) || 0;
    clientMap.set(p.client_name, current + (p.arr || 0));
  });
  
  return Array.from(clientMap.entries())
    .map(([client_name, arr]) => ({ client_name, arr }))
    .sort((a, b) => b.arr - a.arr);
}

// Revenue by Agent Type
function getRevenueByAgentType(features: Feature[]) {
  const agentMap = new Map<string, { total_arr: number; feature_count: number }>();
  
  features.filter(f => f.arr_impact > 0).forEach(f => {
    const agent = f.agent_type || 'Unspecified';
    const current = agentMap.get(agent) || { total_arr: 0, feature_count: 0 };
    agentMap.set(agent, {
      total_arr: current.total_arr + f.arr_impact,
      feature_count: current.feature_count + 1
    });
  });
  
  return Array.from(agentMap.entries())
    .map(([agent_type, data]) => ({ agent_type, ...data }))
    .sort((a, b) => b.total_arr - a.total_arr);
}
```
