```typescript
export const effortLookup: Record<string, number> = {
  'S': 2,   // Small = 2 weeks
  'M': 4,   // Medium = 4 weeks
  'L': 8,   // Large = 8 weeks
  'XL': 12  // Extra Large = 12 weeks
};

// Helper function to convert T-shirt size to weeks
export const getEffortWeeks = (tshirtSize: string | null): number => {
  if (!tshirtSize) return 4; // Default to Medium
  return effortLookup[tshirtSize.toUpperCase()] || 4;
};
```

### 1.4 Priority Score Formula

```typescript
/**
 * Priority Score Formula:
 * (ARR × Replicability × Conversion%) / (Effort_Weeks × Confidence)
 * 
 * Tier Assignment:
 * - Tier 0: Emergency (Revenue_at_Risk = "Yes")
 * - Tier 1: Fast Track (Score >= 100)
 * - Tier 2: Standard Delivery (Score 50-99)
 * - Tier 3: Custom Engagement (Score 10-49)
 * - Tier 4: Backlog (Score < 10)
 */
export const calculatePriorityScore = (feature: Feature): number => {
  // Default ARR to $10M for platform features
  const arr = feature.Primary_Client === 'All' && !feature.ARR_Amount 
    ? 10000000 
    : feature.ARR_Amount || 0;
  
  const replicability = feature.Replicability_Score || 1;
  const conversion = feature.Conversion_Probability_Pct || 100; // Default 100% for platform
  const effortWeeks = getEffortWeeks(feature.Effort_TShirt_Size);
  const confidence = feature.Confidence_Level || 1;
  
  if (effortWeeks === 0 || confidence === 0) return 0;
  return (arr * replicability * (conversion / 100)) / (effortWeeks * confidence);
};

export const getPriorityTier = (score: number, revenueAtRisk: boolean): string => {
  if (revenueAtRisk) return "Tier 0: Emergency";
  if (score >= 100) return "Tier 1: Fast Track";
  if (score >= 50) return "Tier 2: Standard Delivery";
  if (score >= 10) return "Tier 3: Custom Engagement";
  return "Tier 4: Backlog";
};
```
